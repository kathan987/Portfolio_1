Add src/hooks/useReveal.js (IntersectionObserver):

import {useEffect, useRef} from "react";
export default function useReveal(cls="in-view", options={threshold:.2}){
  const ref = useRef(null);
  useEffect(()=>{ const el=ref.current; if(!el) return;
    const io=new IntersectionObserver(([e])=>{ if(e.isIntersecting) el.classList.add(cls); }, options);
    io.observe(el); return ()=>io.disconnect();
  },[]); return ref;
}
